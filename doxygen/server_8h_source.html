215  bool AddConnection(std::string strAddress, uint16_t nPort)
216  {
217  /* Initialize DDOS Protection for Incoming IP Address. */
218  BaseAddress addrConnect(strAddress, nPort);
219 
220  /* Create new DDOS Filter if Needed. */
221  if(!DDOS_MAP.count(addrConnect))
222  DDOS_MAP[addrConnect] = new DDOS_Filter(DDOS_TIMESPAN);
223 
224  /* DDOS Operations: Only executed when DDOS is enabled. */
225  if((fDDOS && DDOS_MAP[addrConnect]->Banned()))
226  return false;
227 
228  /* Find a balanced Data Thread to Add Connection to. */
229  int32_t nThread = FindThread();
230  if(nThread < 0)
231  return false;
232 
233  /* Select the proper data thread. */
234  DataThread<ProtocolType> *dt = DATA_THREADS[nThread];
235  if(!dt)
236  return false;
237 
238  /* Attempt the connection. */
239  if(!dt->AddConnection(strAddress, nPort, DDOS_MAP[addrConnect]))
240  return false;
241 
242  return true;
243  }

430  void Manager()
431  {
432  /* Wait for data threads to startup. */
433  while(DATA_THREADS.size() < MAX_THREADS)
434  runtime::sleep(1000);
435 
436  /* Address to select. */
437  BaseAddress addr;
438 
439  /* Loop connections. */
440  while(!fDestruct.load())
441  {
442  runtime::sleep(1000);
443 
444  /* Assume the connect state is in a failed state. */
445  uint8_t state = static_cast<uint8_t>(ConnectState::FAILED);
446 
447  /* Pick a weighted random priority from a sorted list of addresses. */
448  if(pAddressManager == nullptr)
449  continue;
450 
451  if(pAddressManager->StochasticSelect(addr))
452  {
453  /* Check for connect to self. */
454  if(addr.ToStringIP() == addrThisNode.ToStringIP())
455  {
456  runtime::sleep(1000);
457  continue;
458  }
459 
460  /* Attempt the connection. */
461  debug::log(3, FUNCTION, ProtocolType::Name(), " Attempting Connection ", addr.ToString());
462 
463  /* Attempt the connection. */
464  if(AddConnection(addr.ToStringIP(), addr.GetPort()))
465  {
466  state = static_cast<uint8_t>(ConnectState::CONNECTED);
467 
468  runtime::sleep(nSleepTime);
469  }
470 
471  /* Update the address state. */
472  pAddressManager->AddAddress(addr, state);
473 
474  debug::log(3, FUNCTION, pAddressManager->ToString());
475  }
476  }

527  void ListeningThread(bool fIPv4)
528  {
529  int32_t hListenSocket;
530  SOCKET hSocket;
531  BaseAddress addr;
532  socklen_t len_v4 = sizeof(struct sockaddr_in);
533  socklen_t len_v6 = sizeof(struct sockaddr_in6);
534 
535  /* End the listening thread if LLP set to not listen. */
536  if(!fLISTEN)
537  return;
538 
539  /* Bind the Listener. */
540  if(!BindListenPort(hListenSocket, fIPv4))
541  return;
542 
543  /* Don't listen until all data threads are created. */
544  while(DATA_THREADS.size() < MAX_THREADS)
545  runtime::sleep(1000);
546 
547  /* Setup poll objects. */
548  pollfd fds[1];
549  fds[0].events = POLLIN;
550  fds[0].fd = hListenSocket;
551 
552  /* Main listener loop. */
553  while(!fDestruct.load())
554  {
555  if (hListenSocket != INVALID_SOCKET)
556  {
557  /* Poll the sockets. */
558  int nPoll = poll(&fds[0], 1, 100);
559  if(nPoll < 0)
560  continue;
561 
562  if(!(fds[0].revents & POLLIN))
563  continue;
564 
565  if(fIPv4)
566  {
567  struct sockaddr_in sockaddr;
568 
569  hSocket = accept(hListenSocket, (struct sockaddr*)&sockaddr, &len_v4);
570  if (hSocket != INVALID_SOCKET)
571  addr = BaseAddress(sockaddr);
572  }
573  else
574  {
575  struct sockaddr_in6 sockaddr;
576 
577  hSocket = accept(hListenSocket, (struct sockaddr*)&sockaddr, &len_v6);
578  if (hSocket != INVALID_SOCKET)
579  addr = BaseAddress(sockaddr);
580  }
581 
582  if (hSocket == INVALID_SOCKET)
583  {
584  if (GetLastError() != WSAEWOULDBLOCK)
585  debug::error("socket error accept failed: ", GetLastError());
586  }
587  else
588  {
589  /* Create new DDOS Filter if Needed. */
590  if(!DDOS_MAP.count(addr))
591  DDOS_MAP[addr] = new DDOS_Filter(DDOS_TIMESPAN);
592 
593  /* DDOS Operations: Only executed when DDOS is enabled. */
594  if((fDDOS && DDOS_MAP[addr]->Banned()))
595  {
596  debug::log(3, FUNCTION, "Connection Request ", addr.ToString(), " refused... Banned.");
597  close(hSocket);
598 
599  continue;
600  }
601 
602  Socket_t sockNew(hSocket, addr);
603 
604  int32_t nThread = FindThread();
605  if(nThread < 0)
606  {
607  debug::error(FUNCTION, "Server has no spare connection capacity... dropping");
608  sockNew.Close();
609 
610  continue;
611  }
612 
613  DataThread<ProtocolType> *dt = DATA_THREADS[nThread];
614  if(!dt)
615  continue;
616 
617  dt->AddConnection(sockNew, DDOS_MAP[addr]);
618  debug::log(3, FUNCTION, "Accepted Connection ", addr.ToString(), " on port ", PORT);
619  }
620  }
621  }
622  }

635  bool BindListenPort(int32_t & hListenSocket, bool fIPv4 = true)
636  {
637  std::string strError = "";
638  int32_t nOne = 1;
639 
640  #ifdef WIN32
641  // Initialize Windows Sockets
642  WSADATA wsadata;
643  int32_t ret = WSAStartup(MAKEWORD(2, 2), &wsadata);
644  if (ret != NO_ERROR)
645  {
646  debug::error("TCP/IP socket library failed to start (WSAStartup returned error )", ret);
647  return false;
648  }
649  #endif
650 
651  /* Create socket for listening for incoming connections */
652  hListenSocket = socket(fIPv4 ? AF_INET : AF_INET6, SOCK_STREAM, IPPROTO_TCP);
653  if (hListenSocket == INVALID_SOCKET)
654  {
655  debug::error("Couldn't open socket for incoming connections (socket returned error )", GetLastError());
656  return false;
657  }
658 
659  /* Different way of disabling SIGPIPE on BSD */
660  #ifdef SO_NOSIGPIPE
661  setsockopt(hListenSocket, SOL_SOCKET, SO_NOSIGPIPE, (void*)&nOne, sizeof(int32_t));
662  #endif
663 
664 
665  /* Allow binding if the port is still in TIME_WAIT state after the program was closed and restarted. Not an issue on windows. */
666  #ifndef WIN32
667  setsockopt(hListenSocket, SOL_SOCKET, SO_REUSEADDR, (void*)&nOne, sizeof(int32_t));
668  setsockopt(hListenSocket, SOL_SOCKET, SO_REUSEPORT, (void*)&nOne, sizeof(int32_t));
669  #endif
670 
671 
672  /* The sockaddr_in structure specifies the address family, IP address, and port for the socket that is being bound */
673  if(fIPv4)
674  {
675  struct sockaddr_in sockaddr;
676  memset(&sockaddr, 0, sizeof(sockaddr));
677  sockaddr.sin_family = AF_INET;
678  sockaddr.sin_addr.s_addr = INADDR_ANY; // bind to all IPs on this computer
679  sockaddr.sin_port = htons(PORT);
680  if (::bind(hListenSocket, (struct sockaddr*)&sockaddr, sizeof(sockaddr)) == SOCKET_ERROR)
681  {
682  int32_t nErr = GetLastError();
683  if (nErr == WSAEADDRINUSE)
684  debug::error("Unable to bind to port ", ntohs(sockaddr.sin_port), " on this computer. Address already in use.");
685  else
686  debug::error("Unable to bind to port ", ntohs(sockaddr.sin_port), " on this computer (bind returned error )", nErr);
687 
688  return false;
689  }
690 
691  debug::log(0, FUNCTION,"(v4) Bound to port ", ntohs(sockaddr.sin_port));
692  }
693  else
694  {
695  struct sockaddr_in6 sockaddr;
696  memset(&sockaddr, 0, sizeof(sockaddr));
697  sockaddr.sin6_family = AF_INET6;
698  sockaddr.sin6_addr = IN6ADDR_ANY_INIT; // bind to all IPs on this computer
699  sockaddr.sin6_port = htons(PORT);
700  if (::bind(hListenSocket, (struct sockaddr*)&sockaddr, sizeof(sockaddr)) == SOCKET_ERROR)
701  {
702  int32_t nErr = GetLastError();
703  if (nErr == WSAEADDRINUSE)
704  debug::error("Unable to bind to port ", ntohs(sockaddr.sin6_port), " on this computer. Address already in use.");
705  else
706  debug::error("Unable to bind to port ", ntohs(sockaddr.sin6_port), " on this computer (bind returned error )", nErr);
707 
708  return false;
709  }
710 
711  debug::log(0, FUNCTION, "(v6) Bound to port ", ntohs(sockaddr.sin6_port));
712  }
713 
714  /* Listen for incoming connections */
715  if (listen(hListenSocket, SOMAXCONN) == SOCKET_ERROR)
716  {
717  debug::error("Listening for incoming connections failed (listen returned error )", GetLastError());
718  return false;
719  }
720 
721  return true;
