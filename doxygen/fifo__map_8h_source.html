46 template<class Key>
47 class fifo_map_compare
48 {
49  public:
51  fifo_map_compare(
52  std::unordered_map<Key, std::size_t>* keys,
53  std::size_t timestamp = 1)
54  :
55  m_timestamp(timestamp),
56  m_keys(keys)
57  {}
58 
63  bool operator()(const Key& lhs, const Key& rhs) const
64  {
65  // look up timestamps for both keys
66  const auto timestamp_lhs = m_keys->find(lhs);
67  const auto timestamp_rhs = m_keys->find(rhs);
68 
69  if (timestamp_lhs == m_keys->end())
70  {
71  // timestamp for lhs not found - cannot be smaller than for rhs
72  return false;
73  }
74 
75  if (timestamp_rhs == m_keys->end())
76  {
77  // timestamp for rhs not found - timestamp for lhs is smaller
78  return true;
79  }
80 
81  // compare timestamps
82  return timestamp_lhs->second < timestamp_rhs->second;
83  }
84 
85  void add_key(const Key& key)
86  {
87  m_keys->insert({key, m_timestamp++});
88  }
89 
90  void remove_key(const Key& key)
91  {
92  m_keys->erase(key);
93  }
94 
95  private:
98  template <
99  class MapKey,
100  class MapT,
101  class MapCompare,
102  class MapAllocator
103  > friend class fifo_map;
104 
105  private:
107  std::size_t m_timestamp = 1;
108 
110  std::unordered_map<Key, std::size_t>* m_keys = nullptr;
111 };
